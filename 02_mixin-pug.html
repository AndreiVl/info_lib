	div.services-big(class!=attributes.class style="background-image: url(../images/"+data.imgeBg+".jpg);")
		div.services-big__bg
			div.services-big__text
				include ../services/services.pug
				+startItem({icon: data.icon, title: data.title , text: data.text})

+startBig({
	imgeBg: '',
	icon: '',
	title: '',
	text: ''
})

 mixin startReviews(data)
	div.reviews-list(class!=attributes.class)
		div.container
			div.row.justify-content-center
				- for (var i = 0; i < data.length; i++)
					div.reviews-list__item.col-12.col-md-3
						+#{data[i].name}(data[i].arg)

+startReviews([
	{
		name: 'startBig',
		arg: {
			imgeBg: ''
		}
	}
])

 mixin startValue(data)
	div.start-valueclass!=attributes.class)
		p.text.text--16.text--bold.compound-value__title!=data.title
		- for (var i = 0; i < data.items.length; i++)
			div.compound-value__component
				p.text.text--14.compound-value__dashe 
					span!=data.items[i]['name']
				p.text.text--14.compound-value__number!=data.items[i]['number']

+startValue({
	title: ''.
	items: [
		{
			name: '',
			number: ''
		}
	]
})

Миксин с условиями

 mixin start(data)
	div.start(class!=attributes.class)
		if(data.tag == 'h1')
			h1.start__title=data.text

		if(data.tag == 'h2')
			h2.start__title=data.text

+start({tag: 'h1', text: ''})
+start({tag: 'h2', text: ''})

Миксин социальных кнопок

 mixin social(data)
	ul.social(class!=attributes.class)
		- for (var i = 0; i < data.length; i++)
			li.social__item
				a.social__link(href=data[i]['url'])
					span.visually-hidden=data[i]['text']
					+icon(data[i]['icon'], 'icon--32')
+social([
	{url: '#', text: 'linkidin', icon: 'icon--32-linkidin'},
	{url: '#', text: 'twiter', icon: 'icon--32-twiter'},
	{url: '#', text: 'facebook', icon: 'icon--32-facebook'}
])

 mixin menu(data)
	nav.menu(class!=attributes.class)
		p.h4!=data.title

			- for (var i = 0; i < data.items.length; i++)
				li.menu__item
					+item-menu({icon: data.items[i]['icon'], text: data.items[i]['text']})

	title: '', 
	items: [

	]
